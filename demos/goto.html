Goto extension

The goto extension provides a method for programmatically scrolling to individual child elements. No specific options are required when the Sidescroller is initilized.

Skip to the fifth slide

<script src="src/extensions/overthrow-sidescroller.goto.js"></script>

<script>
var scrollers = document.querySelectorAll( ".overthrow-enabled .sidescroll-goto" );
overthrow.sidescroller(scrollers);

document.querySelector( "a.goto" ).onclick = function( e ){
	overthrow.sidescroller( document.querySelectorAll( ".overthrow-enabled .sidescroll-nextprev" ), "goTo", 4 ); // Slide numbers are zero-indexed.
	e.preventDefault();
}
</script>
